def mse(true: torch.Tensor, predict: torch.Tensor):

    true : torch.Tensor
        The ground truth values.
    predict : torch.Tensor
        The predicted values.

    Returns
    -------
    torch.Tensor
        The computed mean squared error.
    """
    return _mse(true, predict)
